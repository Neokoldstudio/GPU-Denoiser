76Window fabrique_window(char *nom_fen,int x,int y,int width,int height,int zoom)
77{
78 Window win;
79 XSizeHints size_hints;
80 XWMHints wm_hints;
81 XClassHint class_hints;
82 XTextProperty windowName, iconName;
83
84 char *name=nom_fen;
85
86 if(zoom<0) { width/=-zoom; height/=-zoom; }
87 if(zoom>0) { width*=zoom; height*=zoom; }
88
89 win=XCreateSimpleWindow(display,root,x,y,width,height,1,0,255);
90
91 size_hints.flags=PPosition|PSize|PMinSize;
92 size_hints.min_width=width;
93 size_hints.min_height=height;
94
95 XStringListToTextProperty(&name,1,&windowName);
96 XStringListToTextProperty(&name,1,&iconName);
97 wm_hints.initial_state=NormalState;
98 wm_hints.input=True;
99 wm_hints.flags=StateHint|InputHint;
100 class_hints.res_name=nom_fen;
101 class_hints.res_class=nom_fen;
102
103 XSetWMProperties(display,win,&windowName,&iconName,
104 NULL,0,&size_hints,&wm_hints,&class_hints);
105
106 gc=XCreateGC(display,win,0,NULL);
107
108 XSelectInput(display,win,ExposureMask|KeyPressMask|ButtonPressMask|
109 ButtonReleaseMask|ButtonMotionMask|PointerMotionHintMask|
110 StructureNotifyMask);
111
112 XMapWindow(display,win);
113 return(win);
114}

116/****************************************************************************/
117/* CREE_XIMAGE2() */
118/* Crée une XImage à partir d'un tableau de float */
119/* L'image peut subir un zoom. */
120/****************************************************************************/
121XImage* cree_Ximage2(float** mat,int z,int length,int width)
122{
123 int lgth,wdth,lig,col,zoom_col,zoom_lig;
124 float somme;
125 unsigned char pix;
126 unsigned char* dat;
127 XImage* imageX;
128
129 /*Zoom positiv*/
130 /*------------*/
131 if (z>0)
132 {
133 lgth=length*z;
134 wdth=width*z;
135
136 dat=(unsigned char*)malloc(lgth*(wdth*4)*sizeof(unsigned char));
137 if (dat==NULL)
138 { printf("Impossible d'allouer de la memoire.");
139 exit(-1); }
140
141 for(lig=0;lig<lgth;lig=lig+z) for(col=0;col<wdth;col=col+z)
142 {
143 pix=(unsigned char)mat[lig/z][col/z];
144 for(zoom_lig=0;zoom_lig<z;zoom_lig++) for(zoom_col=0;zoom_col<z;zoom_col++)
145 {
146 dat[((lig+zoom_lig)*wdth*4)+((4*(col+zoom_col))+0)]=pix;
147 dat[((lig+zoom_lig)*wdth*4)+((4*(col+zoom_col))+1)]=pix;
148 dat[((lig+zoom_lig)*wdth*4)+((4*(col+zoom_col))+2)]=pix;
149 dat[((lig+zoom_lig)*wdth*4)+((4*(col+zoom_col))+3)]=pix;
150 }
151 }
152 } /*--------------------------------------------------------*/
153
154 /*Zoom negatifv*/
155 /*------------*/
156 else
157 {
158 z=-z;
159 lgth=(length/z);
160 wdth=(width/z);
161
162 dat=(unsigned char*)malloc(lgth*(wdth*4)*sizeof(unsigned char));
163 if (dat==NULL)
164 { printf("Impossible d'allouer de la memoire.");
165 exit(-1); }
166
167 for(lig=0;lig<(lgth*z);lig=lig+z) for(col=0;col<(wdth*z);col=col+z)
168 {
169 somme=0.0;
170 for(zoom_lig=0;zoom_lig<z;zoom_lig++)
171 { for(zoom_col=0;zoom_col<z;zoom_col++)
172 somme+=mat[lig+zoom_lig][col+zoom_col]; }
173
174 somme/=(z*z);
175 dat[((lig/z)*wdth*4)+((4*(col/z))+0)]=(unsigned char)somme;
176 dat[((lig/z)*wdth*4)+((4*(col/z))+1)]=(unsigned char)somme;
177 dat[((lig/z)*wdth*4)+((4*(col/z))+2)]=(unsigned char)somme;
178 dat[((lig/z)*wdth*4)+((4*(col/z))+3)]=(unsigned char)somme;
179 }
180 } /*--------------------------------------------------------*/
181
182 imageX=XCreateImage(display,visual,depth,ZPixmap,0,(char*)dat,wdth,lgth,16,wdth*4);
183 return (imageX);
184}
