212/*
213Short Discrete Cosine Transform
214 data length :8x8, 16x16
215 method :row-column, radix 4 FFT
216functions
217 ddct8x8s : 8x8 DCT
218 ddct16x16s: 16x16 DCT
219function prototypes
220 void ddct8x8s(int isgn, double **a);
221 void ddct16x16s(int isgn, double **a);
222*/
223
224
225/*
226-------- 8x8 DCT (Discrete Cosine Transform) / Inverse of DCT --------
227 [definition]
228 <case1> Normalized 8x8 IDCT
229 C[k1][k2] = (1/4) * sum_j1=0^7 sum_j2=0^7
230 a[j1][j2] * s[j1] * s[j2] *
231 cos(pi*j1*(k1+1/2)/8) *
232 cos(pi*j2*(k2+1/2)/8), 0<=k1<8, 0<=k2<8
233 (s[0] = 1/sqrt(2), s[j] = 1, j > 0)
234 <case2> Normalized 8x8 DCT
235 C[k1][k2] = (1/4) * s[k1] * s[k2] * sum_j1=0^7 sum_j2=0^7
236 a[j1][j2] *
237 cos(pi*(j1+1/2)*k1/8) *
238 cos(pi*(j2+1/2)*k2/8), 0<=k1<8, 0<=k2<8
239 (s[0] = 1/sqrt(2), s[j] = 1, j > 0)
240 [usage]
241 <case1>
242 ddct8x8s(1, a);
243 <case2>
244 ddct8x8s(-1, a);
245 [parameters]
246 a[0...7][0...7] :input/output data (double **)
247 output data
248 a[k1][k2] = C[k1][k2], 0<=k1<8, 0<=k2<8
249*/
250
251
252/* Cn_kR = sqrt(2.0/n) * cos(pi/2*k/n) */
253/* Cn_kI = sqrt(2.0/n) * sin(pi/2*k/n) */
254/* Wn_kR = cos(pi/2*k/n) */
255/* Wn_kI = sin(pi/2*k/n) */
256#define C8_1R 0.49039264020161522456
257#define C8_1I 0.09754516100806413392
258#define C8_2R 0.46193976625564337806
259#define C8_2I 0.19134171618254488586
260#define C8_3R 0.41573480615127261854
261#define C8_3I 0.27778511650980111237
262#define C8_4R 0.35355339059327376220
263#define W8_4R 0.70710678118654752440
264
265void ddct8x8s(int isgn, float **a)
266{
267 int j;
268 float x0r, x0i, x1r, x1i, x2r, x2i, x3r, x3i;
269 float xr, xi;
270
271 if (isgn < 0) {
272 for (j = 0; j <= 7; j++) {
273 x0r = a[0][j] + a[7][j];
274 x1r = a[0][j] - a[7][j];
275 x0i = a[2][j] + a[5][j];
276 x1i = a[2][j] - a[5][j];
277 x2r = a[4][j] + a[3][j];
278 x3r = a[4][j] - a[3][j];
279 x2i = a[6][j] + a[1][j];
280 x3i = a[6][j] - a[1][j];
281 xr = x0r + x2r;
282 xi = x0i + x2i;
283 a[0][j] = C8_4R * (xr + xi);
284 a[4][j] = C8_4R * (xr - xi);
285 xr = x0r - x2r;
286 xi = x0i - x2i;
287 a[2][j] = C8_2R * xr - C8_2I * xi;
288 a[6][j] = C8_2R * xi + C8_2I * xr;
289 xr = W8_4R * (x1i - x3i);
290 x1i = W8_4R * (x1i + x3i);
291 x3i = x1i - x3r;
292 x1i += x3r;
293 x3r = x1r - xr;
294 x1r += xr;
295 a[1][j] = C8_1R * x1r - C8_1I * x1i;
296 a[7][j] = C8_1R * x1i + C8_1I * x1r;
297 a[3][j] = C8_3R * x3r - C8_3I * x3i;
298 a[5][j] = C8_3R * x3i + C8_3I * x3r;
299 }
300 for (j = 0; j <= 7; j++) {
301 x0r = a[j][0] + a[j][7];
302 x1r = a[j][0] - a[j][7];
303 x0i = a[j][2] + a[j][5];
304 x1i = a[j][2] - a[j][5];
305 x2r = a[j][4] + a[j][3];
306 x3r = a[j][4] - a[j][3];
307 x2i = a[j][6] + a[j][1];
308 x3i = a[j][6] - a[j][1];
309 xr = x0r + x2r;
310 xi = x0i + x2i;
311 a[j][0] = C8_4R * (xr + xi);
312 a[j][4] = C8_4R * (xr - xi);
313 xr = x0r - x2r;
314 xi = x0i - x2i;
315 a[j][2] = C8_2R * xr - C8_2I * xi;
316 a[j][6] = C8_2R * xi + C8_2I * xr;
317 xr = W8_4R * (x1i - x3i);
318 x1i = W8_4R * (x1i + x3i);
319 x3i = x1i - x3r;
320 x1i += x3r;
321 x3r = x1r - xr;
322 x1r += xr;
323 a[j][1] = C8_1R * x1r - C8_1I * x1i;
324 a[j][7] = C8_1R * x1i + C8_1I * x1r;
325 a[j][3] = C8_3R * x3r - C8_3I * x3i;
326 a[j][5] = C8_3R * x3i + C8_3I * x3r;
327 }
328 } else {
329 for (j = 0; j <= 7; j++) {
330 x1r = C8_1R * a[1][j] + C8_1I * a[7][j];
331 x1i = C8_1R * a[7][j] - C8_1I * a[1][j];
332 x3r = C8_3R * a[3][j] + C8_3I * a[5][j];
333 x3i = C8_3R * a[5][j] - C8_3I * a[3][j];
334 xr = x1r - x3r;
335 xi = x1i + x3i;
336 x1r += x3r;
337 x3i -= x1i;
338 x1i = W8_4R * (xr + xi);
339 x3r = W8_4R * (xr - xi);
340 xr = C8_2R * a[2][j] + C8_2I * a[6][j];
341 xi = C8_2R * a[6][j] - C8_2I * a[2][j];
342 x0r = C8_4R * (a[0][j] + a[4][j]);
343 x0i = C8_4R * (a[0][j] - a[4][j]);
344 x2r = x0r - xr;
345 x2i = x0i - xi;
346 x0r += xr;
347 x0i += xi;
348 a[0][j] = x0r + x1r;
349 a[7][j] = x0r - x1r;
350 a[2][j] = x0i + x1i;
351 a[5][j] = x0i - x1i;
352 a[4][j] = x2r - x3i;
353 a[3][j] = x2r + x3i;
354 a[6][j] = x2i - x3r;
355 a[1][j] = x2i + x3r;
356 }
357 for (j = 0; j <= 7; j++) {
358 x1r = C8_1R * a[j][1] + C8_1I * a[j][7];
359 x1i = C8_1R * a[j][7] - C8_1I * a[j][1];
360 x3r = C8_3R * a[j][3] + C8_3I * a[j][5];
361 x3i = C8_3R * a[j][5] - C8_3I * a[j][3];
362 xr = x1r - x3r;
363 xi = x1i + x3i;
364 x1r += x3r;
365 x3i -= x1i;
366 x1i = W8_4R * (xr + xi);
367 x3r = W8_4R * (xr - xi);
368 xr = C8_2R * a[j][2] + C8_2I * a[j][6];
369 xi = C8_2R * a[j][6] - C8_2I * a[j][2];
370 x0r = C8_4R * (a[j][0] + a[j][4]);
371 x0i = C8_4R * (a[j][0] - a[j][4]);
372 x2r = x0r - xr;
373 x2i = x0i - xi;
374 x0r += xr;
375 x0i += xi;
376 a[j][0] = x0r + x1r;
377 a[j][7] = x0r - x1r;
378 a[j][2] = x0i + x1i;
379 a[j][5] = x0i - x1i;
380 a[j][4] = x2r - x3i;
381 a[j][3] = x2r + x3i;
382 a[j][6] = x2i - x3r;
383 a[j][1] = x2i + x3r;
384 }
385 }

394float gaussian_noise(float var,float mean)
395{
396 float noise,theta;
397
398 //Noise generation
399 noise=sqrt(-2*var*log(1.0-((float)rand()/RAND_MAX)));
400 theta=(float)rand()*1.9175345E-4-PI;
401 noise=noise*cos(theta);
402 noise+=mean;
403 if (noise>GREY_LEVEL) noise=GREY_LEVEL;
404 if (noise<0) noise=0;
405 return noise;
406}
